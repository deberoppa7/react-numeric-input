## Install

```bash
npm install --save react-numeric-input
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-numeric-input'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
